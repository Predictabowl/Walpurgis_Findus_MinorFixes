ACTOR LoveAndPieceMyrm_Findus : LoveAndPieceMyrm replaces LoveAndPieceMyrm
{
	States
	{
	OnlyMana:
		TNT1 A 0 A_ChangeFlag("INVENTORY.QUIET", TRUE)  
		TNT1 A 0 A_JumpIfInventory("HexenCounter",1,"OnlyMana3")
		Goto OnlyManaDH
	OnlyManaDH:
		TNT1 A 0 {
			if(CountInv("WhetStone", AAPTR_PLAYER1) < 1){
				A_SpawnItemEx("UpgraderItem",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
			} else {
				A_SpawnItemEx("SmallGift",0,0,0,0,0,0,0,SXF_TRANSFERPOINTERS|SXF_TRANSFERAMBUSHFLAG|SXF_TRANSFERSPECIAL|SXF_CLEARCALLERTID|SXF_CLEARCALLERSPECIAL|SXF_SETMASTER,0,tid);
			}
		}
		TNT1 A 0 A_Remove(AAPTR_DEFAULT,RMVF_EVERYTHING)
		Stop
	}
}
